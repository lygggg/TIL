const cuttingRollcake = (topping) => {

let result = 0;

const map = new Map();

const set1 = new Set();

const set2 = new Set();

for (let i = 0; i < topping.length; i++) {

set2.add(topping[i]);

if (map.has(topping[i])) {

map.set(topping[i], map.get(topping[i]) + 1);

} else {

map.set(topping[i], 1);

}

}

  

for (let i = 0; i < topping.length; i++) {

if (set2.has(topping[i])) {

map.set(topping[i], map.get(topping[i]) - 1);

set1.add(topping[i]);

}

if (map.get(topping[i]) === 0) {

set2.delete(topping[i]);

}

if (set1.size === set2.size) {

result += 1;

}

}

return result;

};

test("cuttingRollcake", () => {

expect(cuttingRollcake([1, 2, 1, 3, 1, 4, 1, 2])).toBe(2);

expect(cuttingRollcake([1, 2, 3, 1, 4])).toBe(0);

});
