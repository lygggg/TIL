function pickTangerines(k, tangerine) {

let result = 0;

let sum = 0;

const map = new Map();

tangerine.forEach((e, i) => {

if (map.get(e)) {

map.set(e, map.get(e) + 1);

} else {

map.set(e, 1);

}

});

  

const sortArr = [...map].sort((a, b) => b[1] - a[1]);

  

for (let i = 0; i < sortArr.length; i++) {

if (sum >= k) {

return result;

}

sum += sortArr[i][1];

result += 1;

}

}

  

test("pickTangerines", () => {

expect(pickTangerines(6, [1, 3, 2, 5, 4, 5, 2, 3])).toEqual(3);

expect(pickTangerines(4, [1, 3, 2, 5, 4, 5, 2, 3])).toEqual(2);

expect(pickTangerines(2, [1, 1, 1, 1, 2, 2, 2, 3])).toEqual(1);

});
